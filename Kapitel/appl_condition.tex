\section{application condition}


\begin{construction}
Given a rule $r = L \xhookleftarrow{l} K \xhookrightarrow{r} R$   and a constraint $c$ in ANF with $\nl(c) = 1$. 
Let $G$ and $H$ be be graphs. We denote the set of all overlaps of $G$ and $H$ as $\overlay(G,H)$.
We construct the following application conditions: 

\begin{enumerate}
\item if $c = \forall(a_0: C_0 \to C_1, \false)$: 
		$$\bigvee_{Q \in \overlay(L,C_1)}\biggl(\exists(a:L \inj Q, \true) \wedge \shift(\neg \exists (b: R \to Q', \true)\biggr)$$ with $Q' \setminus (R\setminus (K \cup C_1)) = Q \setminus (L\setminus (K \cup C_1))$. 
		
\item if $c = \exists(a_0: C_0 \to C_1, \true)$:
		$$\bigvee_{Q \in \overlay(L,C_1)} \bigvee_{Q' \subset Q} \biggl(\exists(a:L \to Q',\true) \wedge \bigl(\bigvee_{\substack{\overline{Q} \subset Q \\ Q' \subset \overline{Q}}} \shift(\exists(a: R \to \overline{Q}, \true))\biggr)$$ 
		
\end{enumerate}

\end{construction}
